#include <iostream>
using namespace std;

int main()
{
int numero;
  std::cin >> numero;
  for(int i=1; i<=numero; i++)
  {
     if( i%2 ==1)
        {
            std::cout << "1" << std::endl;
        }
         else  
           { 
            
            std::cout << "0" << std::endl;
        }
      
  }
  
    return 0;
}
